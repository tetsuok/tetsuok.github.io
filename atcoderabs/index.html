AtCoder に登録したら解くべき精選過去問 10 問を Scheme で解いてみた

Qiitaで紹介されている10問を Scheme で解いてみました。Gauche 0.9.5 で動作確認しています。

すでに Scheme で解かれた方の記事 もありますが、人それぞれ解き方が違ってもいいと思うのでここに載せてみます。

第 1 問: ABC 086 A - Product (100 点)

(define (is-prod-even? a b)
  (even? (* a b)))

(let* ((a (read))
       (b (read)))
  (if (is-prod-even? a b)
      (print "Even")
      (print "Odd")))

第 2 問: ABC 081 A - Placing Marbles (100 点)

(define (count-marbles lis)
  (fold + 0 lis))

(let* ((a (digit->integer (read-char)))
       (b (digit->integer (read-char)))
       (c (digit->integer (read-char))))
  (print (count-marbles (list a b c))))

第 3 問: ABC 081 B - Shift Only (200 点)

(define (solve lis)
  (if (find odd? lis)
      0
      (+ (solve (map (lambda (x) (div x 2)) lis)) 1)))

(define (read-seq n)
  (if (zero? n)
      '()
      (cons (read) (read-seq (- n 1)))))

(let* ((N (read))
       (A (read-seq N)))
  (print (solve A)))

第 4 問: ABC 087 B - Coins (200 点)

(define (solve a b c v)
  (let ((ans 0))
    (let loop ((k 0) (j 0) (i 0))
      (let ((total (+ (* 500 i) (* 100 j) (* 50 k))))
        (if (= total v)
            (set! ans (+ ans 1)))
        (cond
         ((and (< k c) (<= total v)) (loop (+ k 1) j i))
         ((< j b) (loop 0 (+ j 1) i))
         ((< i a) (loop 0 0 (+ i 1))))))
    ans))

(let* ((a (read))
       (b (read))
       (c (read))
       (x (read)))
  (print (solve a b c x)))

第 5 問: ABC 083 B - Some Sums (200 点)

(define (sum-of-digit x)
  (if (zero? x)
      0
      (+ (modulo x 10) (sum-of-digit (div x 10)))))

(define (solve n a b)
  (fold + 0
        (filter
         (lambda (x)
           (let ((sum (sum-of-digit x)))
             (and (<= a sum) (<= sum b))))
         (iota n 1))))

(let* ((n (read))
       (a (read))
       (b (read)))
  (print (solve n a b)))

第 6 問: ABC 088 B - Card Game for Two (200 点)

(define (play lis)
  (cond
   ((null? lis) 0)
   ((null? (cdr lis)) (car lis))
   (else
    ;; accumulate Alice - Bob
    (+ (- (car lis) (cadr lis)) (play (cddr lis))))))

(define (solve lis)
  (play (sort lis >)))

(define (read-seq n)
  (if (zero? n)
      '()
      (cons (read) (read-seq (- n 1)))))

(let* ((N (read))
       (A (read-seq N)))
  (print (solve A)))

第 7 問: ABC 085 B - Kagami Mochi (200 点)

(define (pick prev lis)
  (if (null? lis)
      '()
      (let ((head (car lis)))
        (if (= prev head)
            (pick head (cdr lis))
            (cons head (pick head (cdr lis)))))))

(define (solve lis)
  (let ((lis2 (sort lis >)))
    ;; +1 is for car
    (+ 1 (length (pick (car lis2) (cdr lis2))))))

(define (read-seq n)
  (if (zero? n)
      '()
      (cons (read) (read-seq (- n 1)))))

(let* ((N (read))
       (d (read-seq N)))
  (print (solve d)))

第 8 問: ABC 085 C - Otoshidama (300 点)

(define (sum-of-bills x y z)
  (+ (* 10000 x) (* 5000 y) (* 1000 z)))

(define (otoshidama n total)
  (let loop ((y 0) (x 0))
    (let ((z (- n x y)))
      (if (= total (sum-of-bills x y z))
          (list x y z)
          (cond
           ((< y (- n x)) (loop (+ y 1) x))
           ((< x n) (loop 0 (+ x 1)))
           (else '(-1 -1 -1)))))))

(let* ((n (read))
       (total (read)))
  (let ((ans (otoshidama n total)))
    (format #t "~a ~a ~a~%" (car ans) (cadr ans) (caddr ans))))

第 9 問: ABC 049 C - Daydream (300 点)

(use srfi-13)

(define (consume s pat)
  (if (null? pat)
      #f
      (let ((pos (string-scan s (car pat))))
        (if (and (number? pos) (zero? pos))
            (substring s (string-length (car pat)) -1)
            (consume s (cdr pat))))))

(define (solve s pat)
  (let ((rem (consume s pat)))
    (cond
     ((eq? #f rem) "NO")
     ((string=? "" rem) "YES")
     (else (solve rem pat)))))

(let ((pat '("dream" "dreamer" "erase" "eraser"))
      (s (read-line)))
  (print (solve (string-reverse s)
                (map (lambda (x) (string-reverse x)) pat))))

第 10 問: ABC 086 C - Traveling (300 点)

(define (manhattan-distance current next)
  (+ (abs (- (car current) (car next)))
     (abs (- (cadr current) (cadr next)))))

(define (parity=? a b)
  (eq? (even? a) (even? b)))

;; NOTE: parity checkは (+ (cadr next-plan) (caddr next-plan)) と next-t の
;; 比較でもOK
(define (traveling current plan-lis)
  (if (null? plan-lis)
      "Yes"
      (let* ((t (car current))
             (next-plan (car plan-lis))
             (next-t (caar plan-lis))
             (distance (manhattan-distance (cdr current) (cdr next-plan))))
        (if (or (> distance (- next-t t))
                (not (parity=? distance (- next-t t))))
            "No"
            (traveling next-plan (cdr plan-lis))))))

(define (read-seq n)
  (if (zero? n)
      '()
      (cons (list (read) (read) (read)) (read-seq (- n 1)))))

(let* ((N (read))
       (plan-lis (read-seq N)))
  (print (traveling '(0 0 0) plan-lis)))
